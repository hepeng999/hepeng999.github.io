let express = require('express')
let fs = require('fs')
let childProcess = require('child_process')
let pathRoot = process.cwd()
//process.cwd() node里面的进程对象
//process。ENV.NODE_ENV:生产环境还是开发
let app = express()
const port = 3000
app.use((req, res, next) => {
  let fileList = fs.readdirSync(pathRoot)
  res.send({ msg: pathRoot, fileList })
})

app.listen(port, () => {
  console.log('http://localhost:3000')
  if (process.platform === 'win32') {
    childProcess.exec('start http://localhost:3000')
  } else {
    childProcess.exec('open http://localhost:3000')
  }
})
